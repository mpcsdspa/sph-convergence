Let's look at the order of convergence by finding the linear approximation of the log-log plots.

In [43]:
x = np.log10(rho.no_of_particles)
y = np.log10(rho.overall_L1_norm)
o = []
rho_new = []
for i in range(len(x)-1):
    o.append((y[i+1] - y[i])/(x[i+1] - x[i]))
    rho_new.append(0.5*(x[i] + x[i+1]))
In [44]:
plt.plot(rho_new, o)
plt.title('Order of Convergence vs no. of particles')
plt.xlabel('No. of particles')
plt.ylabel('Slope of log-log plot')
plt.ylim([-2, 0])
plt.show()
In [8]:
coefficients = []
for i in range(1,6):
    coefficients.append((np.polyfit(np.log10(rho.no_of_particles), np.log10(rho[[i]]), 1))[0])
